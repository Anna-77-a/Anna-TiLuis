using System;

class Program
{
    static void Main()
    {
        Console.WriteLine("SEQUÊNCIA DE FIBONACCI");

        int n = 10;

        int a = 0;
        int b = 1;

        for (int i = 0; i < n; i++)

            Console.WriteLine(a);

            a = b;
            b = proximo;
        }
    }
}
